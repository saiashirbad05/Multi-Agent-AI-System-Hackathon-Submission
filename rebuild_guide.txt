     → Example: notion.so/My-Page-abc123def456  →  Page ID = abc123def456

  5. GitHub Personal Access Token (for pushing code)
     → github.com/settings/tokens → Generate new token (classic)
     → Check: repo (full access) → copy token (starts with ghp_)

    --member="serviceAccount:${PROJECT_NUMBER}-compute@developer.gserviceaccount.com" \
    --role="roles/firebase.admin"

  gcloud projects add-iam-policy-binding YOUR_PROJECT_ID \
    --member="serviceAccount:${PROJECT_NUMBER}-compute@developer.gserviceaccount.com" \
    --role="roles/storage.admin"

  gcloud projects add-iam-policy-binding YOUR_PROJECT_ID \
    --member="serviceAccount:${PROJECT_NUMBER}-compute@developer.gserviceaccount.com" \
    --role="roles/iam.serviceAccountTokenCreator"

  ⚠️  ERRORS YOU MAY FACE AND FIXES:

  ERROR: "roles/firestore.user is not supported"
  FIX  : Use roles/datastore.owner instead (Firestore uses Datastore roles)

  ERROR: "roles/firebase.admin~ invalid role name"
  FIX  : You accidentally added a ~ at the end, retype the command carefully

  ERROR: "Service account -compute@developer.gserviceaccount.com does not exist"
  FIX  : Your PROJECT_NUMBER variable is empty. Run this instead:
         PROJECT_NUMBER=YOUR_ACTUAL_NUMBER (e.g. 887740208824)
         Then rerun the binding commands

  ❌  WRONG WAY (caused error):
  PROJECT_NUMBER=$(gcloud projects describe MY_PROJECT --format='value(887740208824)')
  → This uses the number as the format field name which is wrong

  ✅  CORRECT WAY:
  PROJECT_NUMBER=$(gcloud projects describe MY_PROJECT --format='value(projectNumber)')

  import os
  from datetime import datetime, timezone
  from google.cloud import firestore

  PROJECT_ID = os.environ.get("GOOGLE_CLOUD_PROJECT", "your-project-id")

  def get_db():
      return firestore.Client(project=PROJECT_ID)

  def save_task(task_id, title, description, due_date, status="pending"):
      try:
          db = get_db()
          data = {
              "task_id": task_id, "title": title, "description": description,
              "due_date": due_date, "status": status,
              "created_at": datetime.now(timezone.utc).isoformat(),
              "updated_at": datetime.now(timezone.utc).isoformat()
          }
          db.collection("tasks").document(task_id).set(data)
          return {"success": True, "task_id": task_id, "message": f"Task '{title}' saved to Firestore"}
      except Exception as e:
          return {"success": False, "error": str(e)}

  def get_task(task_id):
      try:
          db = get_db()
          doc = db.collection("tasks").document(task_id).get()
          if doc.exists:
              return {"success": True, "task": doc.to_dict()}
          return {"success": False, "message": f"Task {task_id} not found"}
      except Exception as e:
          return {"success": False, "error": str(e)}

  def list_tasks(status_filter=None):
      try:
          db = get_db()
          col = db.collection("tasks")
          if status_filter:
              col = col.where("status", "==", status_filter)
          tasks = [doc.to_dict() for doc in col.stream()]
          return {"success": True, "tasks": tasks, "count": len(tasks)}
      except Exception as e:
          return {"success": False, "tasks": [], "count": 0, "error": str(e)}

  def update_task_status(task_id, new_status):
      try:
          db = get_db()
          ref = db.collection("tasks").document(task_id)
          if not ref.get().exists:
              return {"success": False, "message": f"Task {task_id} not found"}
          ref.update({"status": new_status, "updated_at": datetime.now(timezone.utc).isoformat()})
          return {"success": True, "message": f"Task {task_id} updated to '{new_status}'"}
      except Exception as e:
          return {"success": False, "error": str(e)}

  def save_note(note_id, title, content, tags=None):
      try:
          db = get_db()
          data = {
              "note_id": note_id, "title": title, "content": content,
              "tags": tags or [],
              "created_at": datetime.now(timezone.utc).isoformat()
          }
          db.collection("notes").document(note_id).set(data)
          return {"success": True, "note_id": note_id, "message": f"Note '{title}' saved to Firestore"}
      except Exception as e:
          return {"success": False, "error": str(e)}

  def list_notes():
      try:
          db = get_db()
          notes = [doc.to_dict() for doc in db.collection("notes").stream()]
          return {"success": True, "notes": notes, "count": len(notes)}
      except Exception as e:
          return {"success": False, "notes": [], "count": 0, "error": str(e)}
  PYEOF

  import os, uuid
  from datetime import datetime, timedelta, timezone

  def create_calendar_event(title, description, start_time, duration_minutes=60):
      try:
          from googleapiclient.discovery import build
          from google.auth import default
          credentials, _ = default(scopes=["https://www.googleapis.com/auth/calendar"])
          service = build("calendar", "v3", credentials=credentials)
          start_dt = datetime.fromisoformat(start_time)
          end_dt = start_dt + timedelta(minutes=duration_minutes)
          event = {
              "summary": title, "description": description,
              "start": {"dateTime": start_dt.isoformat(), "timeZone": "Asia/Kolkata"},
              "end": {"dateTime": end_dt.isoformat(), "timeZone": "Asia/Kolkata"}
          }
          result = service.events().insert(calendarId="primary", body=event).execute()
          return {"success": True, "event_id": result["id"], "message": f"Event '{title}' created in Google Calendar"}
      except Exception as e:
          # Firestore fallback — always works even without Calendar OAuth
          from google.cloud import firestore
          db = firestore.Client(project=os.environ.get("GOOGLE_CLOUD_PROJECT"))
          event_id = f"event_{str(uuid.uuid4())[:8]}"
          db.collection("calendar_events").document(event_id).set({
              "event_id": event_id, "title": title, "description": description,
              "start_time": start_time, "duration_minutes": duration_minutes,
              "status": "scheduled", "created_at": datetime.now(timezone.utc).isoformat()
          })
          return {"success": True, "event_id": event_id, "message": f"Event '{title}' scheduled and stored in Firestore", "fallback": True}

  def list_upcoming_events(max_results=10):
      try:
          from googleapiclient.discovery import build
          from google.auth import default
          credentials, _ = default(scopes=["https://www.googleapis.com/auth/calendar"])
          service = build("calendar", "v3", credentials=credentials)
          now = datetime.utcnow().isoformat() + "Z"
          result = service.events().list(
              calendarId="primary", timeMin=now,
              maxResults=max_results, singleEvents=True, orderBy="startTime"
          ).execute()
          events = [{"id": e["id"], "title": e["summary"], "start": e["start"].get("dateTime", e["start"].get("date"))} for e in result.get("items", [])]
          return {"success": True, "events": events, "count": len(events)}
      except Exception as e:
          from google.cloud import firestore
          db = firestore.Client(project=os.environ.get("GOOGLE_CLOUD_PROJECT"))
          events = [doc.to_dict() for doc in db.collection("calendar_events").stream()]
          return {"success": True, "events": events, "count": len(events), "fallback": True}
  PYEOF

  import os
  from notion_client import Client

  def get_notion():
      return Client(auth=os.environ.get("NOTION_API_TOKEN"))

  def create_notion_page(title, content):
      try:
          notion = get_notion()
          page_id = os.environ.get("NOTION_PAGE_ID")
          result = notion.pages.create(
              parent={"page_id": page_id},
              properties={"title": {"title": [{"text": {"content": title}}]}},
              children=[{"object": "block", "type": "paragraph",
                         "paragraph": {"rich_text": [{"type": "text", "text": {"content": content}}]}}]
          )
          return {"success": True, "page_id": result["id"], "url": result.get("url", ""), "message": f"Notion page '{title}' created"}
      except Exception as e:
          return {"success": False, "error": str(e)}

  def list_notion_pages():
      try:
          notion = get_notion()
          page_id = os.environ.get("NOTION_PAGE_ID")
          result = notion.blocks.children.list(block_id=page_id)
          pages = []
          for block in result.get("results", []):
              if block.get("type") == "child_page":
                  pages.append({"id": block["id"], "title": block["child_page"]["title"]})
          return {"success": True, "pages": pages, "count": len(pages)}
      except Exception as e:
          return {"success": False, "pages": [], "count": 0, "error": str(e)}
  PYEOF
